              handleEdit={handleEdit}
              handleDelete={handleDelete}
            />
          )}

          {/* URLs View */}
          {activeView === 'urls' && (
            <UrlsView
              urls={urls}
              handleCreateUrl={handleCreateUrl}
              handleEditUrl={handleEditUrl}
              handleDeleteUrl={handleDeleteUrl}
              handleToggleUrlStatus={handleToggleUrlStatus}
              handleCrawlNow={handleCrawlNow}
            />
          )}

          {/* Crawled Articles View */}
          {activeView === 'crawled-articles' && (
            <CrawledArticlesView
              articles={crawledArticles}
              filteredArticles={filteredCrawledArticles}
              loading={loading}
              statusFilter={crawledStatusFilter}
              setStatusFilter={setCrawledStatusFilter}
            />
          )}
        </div>
      </main>

      {/* Article Form Modal */}
      {showModal && (
        <ArticleFormModal
          article={editingArticle}
          onClose={() => {
            setShowModal(false);
            setEditingArticle(null);
          }}
          onSuccess={() => {
            setShowModal(false);
            setEditingArticle(null);
            loadArticles();
          }}
        />
      )}

      {/* URL Form Modal */}
      {showUrlModal && (
        <UrlFormModal
          url={editingUrl}
          onClose={() => {
            setShowUrlModal(false);
            setEditingUrl(null);
          }}
          onSuccess={(newUrl) => {
            if (editingUrl) {
              setUrls(urls.map((u) => (u.id === editingUrl.id ? newUrl : u)));
              toast.success('Crawl URL updated successfully');
            } else {
              setUrls([...urls, { ...newUrl, id: Date.now().toString(), createdAt: new Date().toISOString(), articlesGenerated: 0 }]);
              toast.success('Crawl URL added successfully');
            }
            setShowUrlModal(false);
            setEditingUrl(null);
          }}
        />
      )}

      {/* Delete Article Confirmation Modal */}
      {deleteModalOpen && (
        <div className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center z-[100] p-4">
          <div className="bg-white rounded-2xl shadow-2xl max-w-md w-full p-6 sm:p-8 transform transition-all animate-in fade-in zoom-in duration-200">
            <div className="flex items-center justify-center w-16 h-16 mx-auto bg-red-100 rounded-full mb-4">
              <Trash2 className="w-8 h-8 text-red-600" />
            </div>
            <h3 className="text-xl sm:text-2xl font-bold text-gray-900 text-center mb-2">
              Delete Article?
            </h3>
            <p className="text-gray-600 text-center mb-6">
              Are you sure you want to delete this article? This action cannot be undone.
            </p>
            <div className="flex gap-3">
              <button
                onClick={cancelDelete}
                className="flex-1 px-6 py-3 bg-gray-100 hover:bg-gray-200 text-gray-700 font-semibold rounded-xl transition-all duration-200 hover:scale-105"
              >
                Cancel
              </button>
              <button
                onClick={confirmDelete}
                className="flex-1 px-6 py-3 bg-gradient-to-r from-red-500 to-red-600 hover:from-red-600 hover:to-red-700 text-white font-semibold rounded-xl transition-all duration-200 hover:scale-105 shadow-lg shadow-red-500/30"
              >
                Delete
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Delete URL Confirmation Modal */}
      {deleteUrlModalOpen && (
        <div className="fixed inset-0 bg-black/60 backdrop-blur-sm flex items-center justify-center z-[100] p-4">
          <div className="bg-white rounded-2xl shadow-2xl max-w-md w-full p-6 sm:p-8 transform transition-all animate-in fade-in zoom-in duration-200">
            <div className="flex items-center justify-center w-16 h-16 mx-auto bg-red-100 rounded-full mb-4">
              <Trash2 className="w-8 h-8 text-red-600" />
            </div>
            <h3 className="text-xl sm:text-2xl font-bold text-gray-900 text-center mb-2">
              Delete Crawl URL?
            </h3>
            <p className="text-gray-600 text-center mb-6">
              Are you sure you want to delete this crawl URL? This action cannot be undone.
            </p>
            <div className="flex gap-3">
              <button
                onClick={cancelDeleteUrl}
                className="flex-1 px-6 py-3 bg-gray-100 hover:bg-gray-200 text-gray-700 font-semibold rounded-xl transition-all duration-200 hover:scale-105"
              >
                Cancel
              </button>
              <button
                onClick={confirmDeleteUrl}
                className="flex-1 px-6 py-3 bg-gradient-to-r from-red-500 to-red-600 hover:from-red-600 hover:to-red-700 text-white font-semibold rounded-xl transition-all duration-200 hover:scale-105 shadow-lg shadow-red-500/30"
              >
                Delete
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}

// Crawled Articles View Component
function CrawledArticlesView({
  articles,
  filteredArticles,
  loading,
  statusFilter,
  setStatusFilter,
}: any) {
  const [selectedArticles, setSelectedArticles] = useState<string[]>([]);

  const handleGenerateArticle = (article: CrawledArticle) => {
    toast.success(`Generating AI article from: ${article.title}`, {
      description: 'This will use the crawled content as context for AI generation.',
    });
    // TODO: Implement actual AI generation logic
  };

  const handleBulkGenerate = () => {
    const selectedPendingArticles = articles.filter(
      (a: CrawledArticle) => selectedArticles.includes(a.id) && a.generationStatus === 'pending'
    );
    toast.success(`Initiating bulk generation for ${selectedPendingArticles.length} articles`, {
      description: 'AI will process selected articles in the background.',
    });
    setSelectedArticles([]);
    // TODO: Implement actual bulk generation logic
  };

  const handleSelectAll = () => {
    const pendingIds = filteredArticles
      .filter((a: CrawledArticle) => a.generationStatus === 'pending')
      .map((a: CrawledArticle) => a.id);
    
    if (selectedArticles.length === pendingIds.length && pendingIds.length > 0) {
      setSelectedArticles([]);
    } else {
      setSelectedArticles(pendingIds);
    }
  };

  const handleToggleArticle = (articleId: string) => {
    setSelectedArticles(prev => 
      prev.includes(articleId) 
        ? prev.filter(id => id !== articleId)
        : [...prev, articleId]
    );
  };

  const pendingCount = articles.filter((a: CrawledArticle) => a.generationStatus === 'pending').length;
  const selectedPendingCount = selectedArticles.filter(id => {
    const article = articles.find((a: CrawledArticle) => a.id === id);
    return article && article.generationStatus === 'pending';
  }).length;

  return (
    <div className="space-y-6">
      {/* Header Actions with Filters */}
      <div className="bg-white rounded-2xl shadow-lg border border-gray-200/50 p-6 hover:shadow-xl transition-shadow duration-300">
        <div className="flex flex-col lg:flex-row justify-between items-start lg:items-center gap-4">
          <div className="flex-1">
            <div className="mb-3">
              <p className="text-sm text-gray-600 font-medium">
                View and manage articles crawled from web sources. Click on pending article rows or use checkboxes to select 2+ articles for bulk generation.
              </p>
              {selectedPendingCount >= 1 && selectedPendingCount < 2 && (
                <p className="text-xs text-yellow-600 font-semibold mt-1 flex items-center gap-1">
                  <span>⚠️</span>
                  <span>Select at least one more article to enable bulk generation (minimum 2 required)</span>
                </p>
              )}
            </div>
            
            {/* Filters */}
            <div className="flex flex-wrap gap-3 items-center">
              <div className="flex items-center gap-2">
                <Filter className="w-4 h-4 text-gray-500" />
                <span className="text-xs font-semibold text-gray-500 uppercase">Status:</span>
              </div>
              <button
                onClick={() => setStatusFilter('all')}
                className={`px-3 py-1.5 rounded-lg text-xs font-semibold transition-all duration-200 ${
                  statusFilter === 'all'
                    ? 'bg-gradient-to-r from-gray-700 to-gray-900 text-white shadow-md'
                    : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
                }`}
              >
                All ({articles.length})
              </button>
              <button
                onClick={() => setStatusFilter('pending')}
                className={`px-3 py-1.5 rounded-lg text-xs font-semibold transition-all duration-200 ${
                  statusFilter === 'pending'
                    ? 'bg-gradient-to-r from-yellow-500 to-yellow-600 text-white shadow-md'
                    : 'bg-yellow-50 text-yellow-700 hover:bg-yellow-100'
                }`}
              >
                Pending ({articles.filter((a: CrawledArticle) => a.generationStatus === 'pending').length})
              </button>
              <button
                onClick={() => setStatusFilter('processing')}
                className={`px-3 py-1.5 rounded-lg text-xs font-semibold transition-all duration-200 ${
                  statusFilter === 'processing'
                    ? 'bg-gradient-to-r from-blue-500 to-blue-600 text-white shadow-md'
                    : 'bg-blue-50 text-blue-700 hover:bg-blue-100'
                }`}
              >
                Processing ({articles.filter((a: CrawledArticle) => a.generationStatus === 'processing').length})
              </button>
              <button
                onClick={() => setStatusFilter('generated')}
                className={`px-3 py-1.5 rounded-lg text-xs font-semibold transition-all duration-200 ${
                  statusFilter === 'generated'
                    ? 'bg-gradient-to-r from-green-500 to-green-600 text-white shadow-md'
                    : 'bg-green-50 text-green-700 hover:bg-green-100'
                }`}
              >
                Generated ({articles.filter((a: CrawledArticle) => a.generationStatus === 'generated').length})
              </button>
            </div>
          </div>
          
          {/* Selection Actions */}
          <div className="flex items-center gap-3">
            {selectedPendingCount > 0 && (
              <div className="px-4 py-2 bg-blue-50 text-blue-700 rounded-xl text-sm font-semibold border border-blue-200">
                {selectedPendingCount} Selected
              </div>
            )}
            <button
              onClick={handleBulkGenerate}
              disabled={selectedPendingCount < 2}
              className={`flex items-center gap-2 px-6 py-3 rounded-xl transition-all duration-200 font-semibold shadow-md whitespace-nowrap ${
                selectedPendingCount >= 2
                  ? 'bg-gradient-to-r from-[#ff4500] to-[#ff6b35] text-white hover:shadow-lg hover:scale-105 cursor-pointer'
                  : 'bg-gray-300 text-gray-500 cursor-not-allowed opacity-60'
              }`}
            >
              <Sparkles className={`w-5 h-5 ${selectedPendingCount >= 2 ? 'animate-pulse' : ''}`} />
              <span>Generate Selected {selectedPendingCount > 0 ? `(${selectedPendingCount})` : ''}</span>
            </button>
            {selectedPendingCount > 0 && (
              <button
                onClick={() => setSelectedArticles([])}
                className="flex items-center gap-2 bg-gray-200 text-gray-700 px-4 py-3 rounded-xl hover:bg-gray-300 transition-all duration-200 font-semibold"
              >
                <X className="w-4 h-4" />
                <span>Clear</span>
              </button>
            )}
          </div>
        </div>
      </div>

      {/* Stats */}
      <div className="grid grid-cols-1 sm:grid-cols-4 gap-4">
        <div className="bg-gradient-to-br from-white to-gray-50 rounded-2xl shadow-lg border border-gray-200/50 p-5 hover:shadow-xl transition-all duration-300 hover:scale-105">
          <h3 className="text-xs font-semibold text-gray-500 uppercase tracking-wide mb-1.5">Total Crawled</h3>
          <p className="text-3xl font-bold bg-gradient-to-r from-gray-900 to-gray-700 bg-clip-text text-transparent">{articles.length}</p>
        </div>
        <div className="bg-gradient-to-br from-yellow-50 to-yellow-100/50 rounded-2xl shadow-lg border border-yellow-200/50 p-5 hover:shadow-xl transition-all duration-300 hover:scale-105">
          <h3 className="text-xs font-semibold text-yellow-700 uppercase tracking-wide mb-1.5">Pending</h3>
          <p className="text-3xl font-bold bg-gradient-to-r from-yellow-600 to-yellow-800 bg-clip-text text-transparent">
            {articles.filter((a: CrawledArticle) => a.generationStatus === 'pending').length}
          </p>
        </div>
        <div className="bg-gradient-to-br from-blue-50 to-blue-100/50 rounded-2xl shadow-lg border border-blue-200/50 p-5 hover:shadow-xl transition-all duration-300 hover:scale-105">
          <h3 className="text-xs font-semibold text-blue-600 uppercase tracking-wide mb-1.5">Processing</h3>
          <p className="text-3xl font-bold bg-gradient-to-r from-blue-600 to-blue-800 bg-clip-text text-transparent">
            {articles.filter((a: CrawledArticle) => a.generationStatus === 'processing').length}
          </p>
        </div>
        <div className="bg-gradient-to-br from-green-50 to-green-100/50 rounded-2xl shadow-lg border border-green-200/50 p-5 hover:shadow-xl transition-all duration-300 hover:scale-105">
          <h3 className="text-xs font-semibold text-green-600 uppercase tracking-wide mb-1.5">Generated</h3>

                className="text-[#ff4500] hover:text-[#e63e00] font-semibold text-sm"

                View all crawled articles

            </div>
          ) : (
            <table className="w-full">

                    <td className="px-3 sm:px-6 py-4 sm:py-5">
                      <div className="flex items-start gap-2 sm:gap-3">
                        <img
                          src={article.image}
                          alt=""
                          className="w-12 h-12 sm:w-20 sm:h-20 object-cover rounded-xl shadow-md flex-shrink-0"
                        />
                        <div className="flex-1 min-w-0">

                            {article.title}
                          </p>
                          <p className="text-xs text-gray-500 line-clamp-1 mb-1.5">
                            {article.excerpt}
                          </p>
                          <span className={`inline-flex items-center px-2 sm:px-2.5 py-0.5 sm:py-1 rounded-full text-xs font-semibold shadow-sm ${
                            article.type === 'news'
                              ? 'bg-gradient-to-r from-blue-500 to-blue-600 text-white'
                              : 'bg-gradient-to-r from-purple-500 to-purple-600 text-white'
                          }`}>
                            {article.type}
                          </span>
                        </div>
                      </div>
                    </td>
                    <td className="px-3 sm:px-6 py-4 sm:py-5 whitespace-nowrap">
                      <span className="inline-flex items-center px-2 sm:px-3 py-1 sm:py-1.5 rounded-full text-xs font-semibold bg-gradient-to-r from-gray-100 to-gray-200 text-gray-800 shadow-sm">
                        {article.category}
                      </span>
                    </td>
                    <td className="px-3 sm:px-6 py-4 sm:py-5 whitespace-nowrap">
                      <span className={`inline-flex items-center gap-1.5 px-2.5 sm:px-3 py-1.5 rounded-full text-xs font-bold shadow-md ${
                        article.generationStatus === 'pending'
                          ? 'bg-gradient-to-r from-yellow-400 to-yellow-500 text-yellow-900'
                          : article.generationStatus === 'processing'
                          ? 'bg-gradient-to-r from-blue-500 to-blue-600 text-white'
                          : 'bg-gradient-to-r from-green-500 to-green-600 text-white'
                      }`}>
                        {article.generationStatus === 'processing' && (
                          <Loader2 className="w-3 h-3 animate-spin" />
                        )}
                        {article.generationStatus === 'generated' && (
                          <Sparkles className="w-3 h-3" />
                        )}
                        <span className="capitalize">{article.generationStatus}</span>
                      </span>
                    </td>
                    <td className="px-3 sm:px-6 py-4 sm:py-5 hidden sm:table-cell">
                      <a 
                        href={article.sourceUrl} 
                        target="_blank" 
                        rel="noopener noreferrer"
                        className="text-xs text-blue-600 hover:text-blue-800 hover:underline font-medium flex items-center gap-1 max-w-[150px] truncate"
                        title={article.sourceUrl}
                      >
                        <LinkIcon className="w-3 h-3 flex-shrink-0" />
                        <span className="truncate">View Source</span>
                      </a>
                    </td>
                    <td className="px-3 sm:px-6 py-4 sm:py-5 whitespace-nowrap hidden lg:table-cell">
                      <div className="flex flex-col gap-1">
                        <span className="text-xs sm:text-sm text-gray-600 font-medium">
                          {new Date(article.crawledAt).toLocaleDateString()}
                        </span>
                        <span className="text-xs text-gray-400">
                          {new Date(article.crawledAt).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}
                        </span>
                      </div>
                    </td>
                    <td className="px-3 sm:px-6 py-4 sm:py-5 whitespace-nowrap text-right">
                      <div className="flex items-center justify-end gap-2">
                        {article.generationStatus === 'pending' ? (
                          <button
                            onClick={() => handleGenerateArticle(article)}
                            className="inline-flex items-center gap-1.5 px-3 sm:px-4 py-2 sm:py-2.5 bg-gradient-to-r from-[#ff4500] to-[#ff6b35] text-white rounded-xl hover:shadow-lg hover:scale-105 transition-all duration-200 font-semibold text-xs sm:text-sm shadow-md group"
                            title="Generate AI Article from this content"
                          >
                            <Sparkles className="w-4 h-4 animate-pulse" />
                            <span className="hidden sm:inline">Generate</span>
                          </button>
                        ) : article.generationStatus === 'processing' ? (
                          <div className="flex items-center gap-2 px-3 sm:px-4 py-2 sm:py-2.5 bg-blue-100 text-blue-700 rounded-xl text-xs sm:text-sm font-semibold">
                            <Loader2 className="w-4 h-4 animate-spin" />
                            <span className="hidden sm:inline">Processing...</span>
                          </div>
                        ) : (
                          <div className="flex items-center gap-2 px-3 sm:px-4 py-2 sm:py-2.5 bg-green-100 text-green-700 rounded-xl text-xs sm:text-sm font-semibold">
                            <Sparkles className="w-4 h-4" />
                            <span className="hidden sm:inline">Completed</span>
                          </div>
                        )}
                      </div>
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          )}
        </div>
      </div>
    </div>
  );
}

// Article Form Modal Component
function ArticleFormModal({
  article,
  onClose,
  onSuccess,
}: {
  article: Article | null;
  onClose: () => void;
  onSuccess: () => void;
}) {
  const [formData, setFormData] = useState({
    type: article?.type || 'news',
    category: article?.category || 'Breaking / Trending',
    title: article?.title || '',
    description: article?.description || '',
    content: article?.content || '',
    image: article?.image || '',
    readTime: article?.readTime || '5 min read',
    aiGenerated: article?.aiGenerated || false,
    featured: article?.featured || false,
    authorName: article?.author.name || 'Admin',
    authorRole: article?.author.role || 'Content Editor',
  });
  const [saving, setSaving] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();

    setSaving(true);

    try {
      const articleData = {
        ...formData,
        author: {
          name: formData.authorName,
          avatar: `https://i.pravatar.cc/40?img=${Math.floor(Math.random() * 70)}`,
          role: formData.authorRole,
        },
      };

      if (article) {
        await articlesApi.updateArticle(article.id, articleData, '');
        toast.success('Article updated successfully');
      } else {
        await articlesApi.createArticle(articleData, '');
        toast.success('Article created successfully');
      }

      onSuccess();
    } catch (error) {
      toast.error(article ? 'Failed to update article' : 'Failed to create article');
    } finally {
      setSaving(false);
    }
  };

  return (
    <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-2 sm:p-4">
      <div className="bg-white rounded-lg shadow-2xl max-w-4xl w-full max-h-[90vh] overflow-y-auto pr-1 [&::-webkit-scrollbar]:w-2 [&::-webkit-scrollbar-track]:bg-transparent [&::-webkit-scrollbar-track]:my-2 [&::-webkit-scrollbar-thumb]:bg-gray-400 [&::-webkit-scrollbar-thumb]:rounded-full [&::-webkit-scrollbar-thumb]:border-2 [&::-webkit-scrollbar-thumb]:border-solid [&::-webkit-scrollbar-thumb]:border-white [&::-webkit-scrollbar-thumb]:hover:bg-gray-500">

          >
            <X className="w-5 h-5 sm:w-6 sm:h-6" />
          </button>

                Category
              </label>
              <select
                value={formData.category}
                onChange={(e) =>
                  setFormData({ ...formData, category: e.target.value })
                }
                className="w-full px-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-[#ff4500] focus:border-transparent outline-none"
              >
